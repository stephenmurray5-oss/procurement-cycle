}
];

const ProcurementCycleAccordion = () => {
const [openId, setOpenId] = useState(null);

const handleClick = (id) => {
setOpenId(openId === id ? null : id);
};

return (
<div>
<h2>Procurement Cycle</h2>
{sections.map((section) => (
<div key={section.id} style={{ marginBottom: "1em", border: "1px solid #ccc", borderRadius: "6px", padding: "10px"}}>
<button
style={{
width: "100%",
textAlign: "left",
padding: "10px",
fontSize: "1rem",
background: "#eaf6ff",
border: "none",
borderRadius: "6px",
cursor: "pointer"
}}
onClick={() => handleClick(section.id)}
>
{section.id}. {section.title}
</button>
{openId === section.id && (
<div style={{ marginTop: "10px", padding: "10px", background: "#f8fdff", borderRadius: "6px"}}>
{section.description}
</div>
)}
</div>
))}
</div>
);
};

export default ProcurementCycleAccordion;
